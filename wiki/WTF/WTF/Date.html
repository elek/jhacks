WTF/Date

Egy masik java.util.Date perverzio, amit Zsoltk tett kozze regebben. Magyon elnagyolva valahogy igy...


import java.sql.Connection;
// ... satobbi

public class DateUtil {
  public static Date getCurrentDate(Connection conn){
    Statement sta = conn.createStatement();
    ResultSet res = sta.executeQuery("select sysdate from dual");
    res.next();
    Date d = res.getDate(1);
    sta.close();
    return d;
  }
}
